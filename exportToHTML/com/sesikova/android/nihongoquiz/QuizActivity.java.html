24   public class QuizActivity extends AppCompatActivity {
25   
26       private static final String TAG = MainActivity.class.getSimpleName();
27   
28       private TextView questionNumber, question;
29       private RadioGroup radioGroup;
30       private RadioButton option1, option2, option3, option4;
31   
32       private List<QuizObject> quizList;
33       private QuizObject quizObject;
34   
35       private int quizCount;
36       private int quizIndex = 0;
37   
38       private ScoreObject scoreObject;
39   
40   
41   
42       @Override
43       protected void onCreate(Bundle savedInstanceState) {
44   
45           setRequestedOrientation(ActivityInfo.SCREEN_ORIENTATION_PORTRAIT);
46           super.onCreate(savedInstanceState);
47           setContentView(R.layout.activity_quiz);
48   
49           String categoryName = getIntent().getExtras().getString("TOPIC_NAME");
50           if(!TextUtils.isEmpty(categoryName)){
51               setTitle("Quiz on " + categoryName);
52           }
53   
54           scoreObject = new ScoreObject();
55   
56           questionNumber = (TextView)findViewById(R.id.question_number);
57           question = (TextView)findViewById(R.id.question);
58   
59           radioGroup = (RadioGroup)findViewById(R.id.answer_options);
60   
61           option1 = (RadioButton)findViewById(R.id.answer_1);
62           option2 = (RadioButton)findViewById(R.id.answer_2);
63           option3 = (RadioButton)findViewById(R.id.answer_3);
64           option4 = (RadioButton)findViewById(R.id.answer_4);
65   
66           int topicId = getIntent().getExtras().getInt("TOPIC_ID");
67           DataBaseQuery query = new DataBaseQuery(QuizActivity.this);
68   
69           quizList = query.getQuizList(topicId);
70   
71           Button nextQuestionButton = (Button)findViewById(R.id.next_quiz);
72   
73           if(quizList.size() > 0){
74               quizCount = quizList.size();
75               quizObject = quizList.get(quizIndex);
76               displayQuizQuestions();
77   
78               assert nextQuestionButton != null;
79               nextQuestionButton.setOnClickListener(new View.OnClickListener() {
80                   @Override
81                   public void onClick(View view) {
82                       int radioButtonId = radioGroup.getCheckedRadioButtonId();
83                       String userSelectedAnswer = selectedAnswerOption(radioButtonId);
84   
85                       if(userSelectedAnswer.equals("")){
86                           Toast.makeText(QuizActivity.this, "You must select an answer " + userSelectedAnswer, Toast.LENGTH_LONG).show();
87                       }else{
88                           //check for the correct answer
89                           boolean isCorrect;
90   
91                           if(quizObject.getAnswer().trim().equals(userSelectedAnswer.trim())){
92                               //set new score
93                               isCorrect = true;
94                               scoreObject.setScore(1);
95                           }else{
96                               isCorrect = false;
97                           }
98   
99   
100  
101                          //add the result
102                          scoreObject.addNewResult(new ResultObject(Integer.toString(quizIndex+1), quizObject.getQuestion(), userSelectedAnswer, quizObject.getAnswer(),isCorrect));
103  
104                          quizIndex++;
105  
106                          // check if there is more question
107                          if(quizIndex >= quizCount){
108                              // Quiz over
109                              Intent quizActivityIntent = new Intent(QuizActivity.this, ResultActivity.class);
110  
111                              Gson gson = new GsonBuilder().create();
112                              final String scoreString = gson.toJson(scoreObject);
113  
114                              quizActivityIntent.putExtra("RESULT_OBJECT", scoreString);
115                              quizActivityIntent.putExtra("QUIZ_SCORE", String.valueOf(scoreObject.getScore()));
116                              quizActivityIntent.putExtra("QUIZ_COUNT", String.valueOf(quizCount));
117  
118                              // compare score and save
119                              double percentageScore = (scoreObject.getScore() * 100) / quizCount ;
120                              Double mDouble = new Double(percentageScore);
121  
122  
123                              startActivity(quizActivityIntent);
124  
125                          }else{
126                              // display new questions
127                              quizObject = quizList.get(quizIndex);
128  
129                              displayQuizQuestions();
130                          }
131                      }
132                  }
133              });
134          }else{
135              option1.setVisibility(View.GONE);
136              option2.setVisibility(View.GONE);
137              option3.setVisibility(View.GONE);
138              option4.setVisibility(View.GONE);
139              nextQuestionButton.setVisibility(View.GONE);
140              Toast.makeText(QuizActivity.this, getString(R.string.no_quiz), Toast.LENGTH_LONG).show();
141          }
142      }
143  
144      private void displayQuizQuestions(){
145          if(quizObject != null){
146              unsetRadioButton();
147              int currentQuestion = quizIndex + 1;
148              questionNumber.setText("Question " + currentQuestion);
149              question.setText(quizObject.getQuestion());
150  
151              String answerOption = quizObject.getOptions();
152              String[] allAnswerOptions = quizObject.convertOptionsToStringArray(answerOption);
153  
154              option1.setText(allAnswerOptions[0]);
155              option2.setText(allAnswerOptions[1]);
156              option3.setText(allAnswerOptions[2]);
157              option4.setText(allAnswerOptions[3]);
158          }
159      }
160  
161      private void unsetRadioButton(){
162          radioGroup.clearCheck();
163          option1.setChecked(false);
164          option2.setChecked(false);
165          option3.setChecked(false);
166          option4.setChecked(false);
167      }
168  
169      private String selectedAnswerOption(int id){
170          String textContent = "";
171  
172          if(id == R.id.answer_1){
173              textContent = option1.getText().toString();
174          }
175          if(id == R.id.answer_2){
176              textContent = option2.getText().toString();
177          }
178          if(id == R.id.answer_3){
179              textContent = option3.getText().toString();
180          }
181          if(id == R.id.answer_4){
182              textContent = option4.getText().toString();
183          }
184          return textContent;
185      }
